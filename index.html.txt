  <script>
    // alert close
    document.querySelectorAll("[data-close]").forEach(btn=>{
      btn.addEventListener("click", ()=> btn.closest(".alert")?.remove());
    });

    // academic close
    document.getElementById("academicClose").addEventListener("click", ()=>{
      document.getElementById("academic").style.display = "none";
    });

    // encrypted toggle
    const enc = document.getElementById("enc");
    const encResult = document.getElementById("encResult");
    enc.addEventListener("click", ()=>{
      const on = enc.classList.toggle("on");
      encResult.classList.toggle("show", on);
      enc.textContent = on ? "DECODED" : "ENCRYPTED: CLICK TO DECODE";
    });

    // audio play/pause (파일 연결하면 작동)
    const bgm = document.getElementById("bgm");
    const playBtn = document.getElementById("playBtn");
    let playing = false;
    playBtn.addEventListener("click", async ()=>{
      try{
        if(!playing){ await bgm.play(); playing = true; playBtn.textContent="PAUSE"; }
        else{ bgm.pause(); playing = false; playBtn.textContent="PLAY"; }
      }catch(e){
        alert("오디오 소스가 없거나 브라우저 정책으로 재생이 제한될 수 있어요.");
      }
    });
